18    @Test
19    public void isValid() throws Exception {
20      String[] args0 = {"12", "2", "0"};
21      assertFalse(new CmdArgument(args0).isValid());
22      String[] args1 = {"a", "2", "0", "test.csv"};
23      assertFalse(new CmdArgument(args1).isValid());
24      String[] args2 = {"-1", "2", "0", "test.csv"};
25      assertFalse(new CmdArgument(args2).isValid());
26      String[] args3 = {"50000000", "2", "0", "test.csv"};
27      assertFalse(new CmdArgument(args3).isValid());
28      String[] args4 = {"12", "b", "0", "test.csv"};
29      assertFalse(new CmdArgument(args4).isValid());
30      String[] args5 = {"12", "-2", "0", "test.csv"};
31      assertFalse(new CmdArgument(args5).isValid());
32      String[] args6 = {"12", "20000", "0", "test.csv"};
33      assertFalse(new CmdArgument(args6).isValid());
34      String[] args7 = {"12", "2", "c", "test.csv"};
35      assertFalse(new CmdArgument(args7).isValid());
36      String[] args8 = {"12", "2", "2", "test.csv"};
37      assertFalse(new CmdArgument(args8).isValid());
38      String[] args9 = {"12", "2", "2", "test"};
39      assertFalse(new CmdArgument(args9).isValid());
40    }
41  
42    @Test
43    public void testEquals() throws Exception {
44      assertFalse(handler.equals(null));
45      assertTrue(handler.equals(handler));
46      assertTrue(handler.equals(new CmdArgument(args)));
47      String[] args0 = {"12", "2", "0"};
48      assertFalse(handler.equals(new CmdArgument(args0)));
49      String[] args1 = {"4", "5", "1", "test.csv"};
50      assertFalse(handler.equals(new CmdArgument(args1)));
51      String[] args2 = {"10", "2", "1", "test.csv"};
52      assertFalse(handler.equals(new CmdArgument(args2)));
53      String[] args3 = {"10", "5", "0", "test.csv"};
54      assertFalse(handler.equals(new CmdArgument(args3)));
55      String[] args4 = {"10", "5", "1", "test1.csv"};
56      assertFalse(handler.equals(new CmdArgument(args4)));
57      String[] args5 = {"4", "5", "1", null};
58      assertTrue((new CmdArgument(args5)).equals(new CmdArgument(args5)));
59    }
