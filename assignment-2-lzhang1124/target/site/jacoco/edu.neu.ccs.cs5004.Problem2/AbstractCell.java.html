package edu.neu.ccs.cs5004.Problem2;

import java.util.Objects;

public abstract class AbstractCell implements Cell {

    protected Boolean isHit;
    protected Boolean hasShip;

    public AbstractCell(Boolean isHit, Boolean hasShip) {
        this.isHit = isHit;
        this.hasShip = hasShip;
    }

    public Boolean getIsHit() {
        return isHit;
    }

    public Boolean getHasShip() {
        return hasShip;
    }

    @Override
    public Cell attack() {
        this.isHit = true;
        return this;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        AbstractCell that = (AbstractCell) o;
        return Objects.equals(isHit, that.isHit) &&
                Objects.equals(hasShip, that.hasShip);
    }

    @Override
    public int hashCode() {

        return Objects.hash(isHit, hasShip);
    }

}
