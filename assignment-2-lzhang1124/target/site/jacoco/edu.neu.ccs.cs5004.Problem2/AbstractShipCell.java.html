package edu.neu.ccs.cs5004.Problem2;

import java.util.Objects;

public abstract class AbstractShipCell extends AbstractCell {
    protected Ship ship;
    protected markSunk markSunk;

    public AbstractShipCell(Boolean isHit, Boolean hasShip, Ship ship, markSunk markSunk) {
        super(isHit, hasShip);
        this.hasShip = true;
        this.ship = ship;
        this.markSunk = markSunk;

    }

    public Ship getShip() {
        return ship;
    }

    public markSunk getMarkSunk() {
        return markSunk;
    }

    @Override
    public String attackResult() {
        if (this.ship.isSunk()) return "Sunk";
        else return "Hit";
    }

    @Override
    public Boolean canPlacedShip() {
        return false;
    }

    @Override
    public Cell attack() {
        this.isHit = true;
        this.ship.hitShip();
        return this;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        if (!super.equals(o)) return false;
        AbstractShipCell that = (AbstractShipCell) o;
        return Objects.equals(ship, that.ship) &&
                Objects.equals(markSunk, that.markSunk);
    }

    @Override
    public int hashCode() {

        return Objects.hash(super.hashCode(), ship, markSunk);
    }

}
