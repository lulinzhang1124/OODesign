package edu.neu.ccs.cs5004.Problem2;

import java.util.Objects;

public abstract class AbstractWaterCell extends AbstractCell {
    protected Boolean hasAdjacentCell;

    public AbstractWaterCell(Boolean isHit, Boolean hasShip, Boolean hasAdjacentCell) {
        super(isHit, hasShip);
        this.hasShip = false;
        this.hasAdjacentCell = hasAdjacentCell;
    }

    public Boolean getHasAdjacentCell() {
        return this.hasAdjacentCell;
    }

    @Override
    public Boolean canPlacedShip() {
        return !(this.hasAdjacentCell);
    }

    @Override
    public String attackResult() {
        return "Miss";
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        if (!super.equals(o)) return false;
        AbstractWaterCell that = (AbstractWaterCell) o;
        return Objects.equals(hasAdjacentCell, that.hasAdjacentCell);
    }

    @Override
    public int hashCode() {

        return Objects.hash(super.hashCode(), hasAdjacentCell);
    }

}
