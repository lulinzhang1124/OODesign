46      @Test
47      public void testEquals() throws Exception {
48          //the variables below are requried only for this test, hence NO need to set it as a field.
49          AbstractShip nullAbsCell = null;
50          AbstractShip yetAnotherAbsCell = new Battleship(4, 1);
51          AbstractShip diffAbsCell1 = new Submarine(2, 1);
52          AbstractShip diffAbsCell2 = new Battleship(4, 0);
53  
54          Assert.assertTrue(absShip.equals(absShip)); // reflexivity
55          Assert.assertTrue(absShip.equals(sameRefAbsCell)); // trivial condition both reference the same object
56          Assert.assertEquals(absShip.equals(sameStateAbsCell), sameStateAbsCell.equals(absShip)); //symmetry
57          Assert.assertEquals(absShip.equals(sameStateAbsCell) &&
58                  sameStateAbsCell.equals(yetAnotherAbsCell), yetAnotherAbsCell.equals(absShip)); //transitivity
59          Assert.assertFalse(absShip.equals(diffAbsCell1));
60          Assert.assertFalse(absShip.equals(diffAbsCell2));
61          Assert.assertFalse(absShip.equals(nullAbsCell));
62      }
