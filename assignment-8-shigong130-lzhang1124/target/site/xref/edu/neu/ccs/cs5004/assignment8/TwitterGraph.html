9   public interface TwitterGraph {
10    /**
11     * Build a graph using a node file and an edge file.

13     * @param file1 node file.
14     * @param file2 edge file.
15     */
16    void buildGraph(String file1, String file2);
17  
18    /**
19     * Gets the total number of users.
20     *
21     * @return total number of users.
22     */
23    Integer totalNumberOfUsers();
24  
25    /**
26     * Gets the list of user that this program need to make recommendations.
27     *
28     * @param processingFlag         - a character from s, e, r.
29     * @param numberOfUsersToProcess - number of users you need to process to make recommendation.
30     * @return the users list.
31     */
32    List<Integer> processUsers(String processingFlag, Integer numberOfUsersToProcess);
33  
34    /**
35     * Gets all user ids.
36     *
37     * @return - a list of all user ids.

41    /**
42     * Gets the set of followers of a given user.
43     *
44     * @param twitterId id number of the user
45     * @return this user's friends list
46     */
47    Set<Integer> getFollowers(Integer twitterId);
48  
49    /**
50     * This method is used to get the follows of a user according to his ID.
51     *
52     * @param twitterId id number of the user
53     * @return this user's follows
54     */
55    Set<Integer> getFollows(Integer twitterId);
56  
57    /**
58     * Gets the influencers of all users based on the criterion given.
59     *
60     * @param criterionOfInfluencer used to define whether a node is an influencer.
61     * @return list of users who are influencers.
62     */
63    List<Integer> getInfluencers(int criterionOfInfluencer);
64  
65  
66    /**
67     * Gets the node information of a given user id.
68     *
69     * @param twitterId id number of the user.
70     * @return user node.
71     */
72    User getNode(Integer twitterId);
