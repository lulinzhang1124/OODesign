21    public void setUp() throws Exception {

23      sameRefEdge = edge;
24      sameStateEdge = Edge.buildFromLine("1,2");
25      yetAnotherEdge = Edge.buildFromLine("1,2");
26      diffEdge = Edge.buildFromLine("3.4");
27      nullEdge = null;
28    }

30    @Test
31    public void testEdge() throws Exception {
32  
33  
34      assertEquals(edge, new Edge(1, 2));
35      assertTrue(Edge.buildFromLine("") == null);
36      assertEquals(edge.hashCode(), new Edge(1, 2).hashCode());
37      assertFalse(edge.equals(new Edge(2, 3)));
38      assertFalse(edge.equals(new Object()));
39    }

51    }

53    @Test
54    public void buildFromLine() {
55      Assert.assertNull(edge.buildFromLine("3"));
56      Assert.assertNull(edge.buildFromLine("3, 4, 5"));
57      Assert.assertEquals(edge, Edge.buildFromLine("1,2"));
58  
59    }

61    @Test
62    public void equals() {
63  
64      Assert.assertTrue(edge.equals(edge));
65      Assert.assertTrue(edge.equals(sameRefEdge));
66      Assert.assertTrue(edge.equals(sameStateEdge));
67      Assert.assertTrue(sameStateEdge.equals(edge));
68      Assert.assertEquals(edge.equals(sameStateEdge), sameStateEdge.equals(edge));
69      Assert.assertEquals(edge.equals(sameStateEdge) && sameStateEdge.equals(yetAnotherEdge), yetAnotherEdge.equals(edge));
70      Assert.assertFalse(edge.equals(diffEdge));
71      Assert.assertFalse(edge.equals(new Integer(1)));
72      Assert.assertFalse(edge.equals(nullEdge));
73  
74    }

77    public void testHashCode() {

80    }
