34    @Test
35    public void equals() {
36      Lawyer nulllawyer = null;
37      Lawyer yetAnotherlawyer = new Lawyer(new Name("Jon", "Snow"), 40,
38          14, "L166Y15", 3);
39      Lawyer difflawyer = new Lawyer(new Name("Julie", "West"), 40,
40          14, "L24Y17", 3);
41      Lawyer difflawyer2 = new Lawyer(new Name("Jon", "Snow"), 35,
42          14, "L166Y15", 3);
43      Lawyer difflawyer4 = new Lawyer(new Name("Jon", "Snow"), 40,
44          15, "L166Y15", 3);
45      Lawyer difflawyer5 = new Lawyer(new Name("Jon", "Snow"), 40,
46          14, "L166Y15", 1);
47      Lawyer difflawyer6 = new Lawyer(new Name("Jon", "Snow"), 40,
48          14, "L166Y18", 3);
49  
50      assertTrue(lawyer.equals(lawyer)); // reflexivity
51      assertTrue(lawyer.equals(sameReflawyer)); // trivial condition both reference the same
52      assertTrue(lawyer.equals(sameStateAslawyer)); // both objects should have the same
53      assertTrue(sameStateAslawyer.equals(lawyer)); //symmetry
54      assertEquals(lawyer.equals(sameStateAslawyer) && sameStateAslawyer.equals
55          (yetAnotherlawyer), yetAnotherlawyer.equals(lawyer)); //transitivity
56      assertFalse(lawyer.equals(difflawyer)); //objects have different state
57      assertFalse(lawyer.equals(difflawyer2));
58      assertFalse(lawyer.equals(difflawyer4));
59      assertFalse(lawyer.equals(difflawyer5));
60      assertFalse(lawyer.equals(difflawyer6));
61      assertFalse(lawyer.equals(nulllawyer)); // lawyer is NOT null
62      assertFalse(lawyer.equals(new Integer(4)));
63    }
