59    /**
60     * Print the the specificShipCell.
61     *
62     * @param specificShipCell the cell to be printed.
63     */
64    @Override
65    public void toConsole(SpecificShipCell specificShipCell) {
66      if (specificShipCell.getShip().isSunk()) {
67        System.out.print(" / ");
68      } else if (specificShipCell.getIsHit()) {
69        System.out.print(" X ");
70      } else {
71        System.out.print(" O ");
72      }
73    }

107   /**
108    * Print the map.
109    *
110    * @param map the map to be printed.
111    */
112   @Override
113   public void toConsole(Map map) {
114 
115     System.out.println();
116     System.out.println("    A   B   C   D   E   F   G   H   I   J"); //alphabetic notation
117     System.out.println(" ###########################################");
118     for (int i = 0; i < Map.ROW; i++) { //rows
119       System.out.print(" #|");
120       for (int j = 0; j < Map.COLUMN; j++) { //columns
121 
122         this.toConsole(map.getMap(Row.values()[i], Column.values()[j]));
123         System.out.print("|");
124       }
125       System.out.println("# " + (i + 1)); //next row
126       if (i != Map.ROW - 1) {
127         System.out.println(" #|---------------------------------------|#");//linebreak between rows
128       } else {
129         System.out.println(" ###########################################");
130       }
131     }
132 
133     System.out.println("    A   B   C   D   E   F   G   H   I   J");//alphabetic notation
134     System.out.println();
135 
136   }
