3   import edu.neu.ccs.cs5004.assignment5.battleship.ships.Ship;
4   import edu.neu.ccs.cs5004.assignment5.battleship.viewer.Printer;

6   /**
7    * Represents a cell.
8    */
9   public interface Cell {
10    /**
11     * Get the attack result for a cell.
12     *
13     * @return a attack result
14     */
15    AttackResult attackResult();
16  
17    /**
18     * Indicate a ship can be placed on this cell or not.
19     *
20     * @return true if a ship can place on this cell, false otherwise
21     */
22    Boolean canPlacedShip();
23  
24    /**
25     * Attack the cell.
26     *
27     * @return the updated cell after the attack
28     */
29    Cell attack();
30  
31    /**
32     * Print the cell.
33     *
34     * @param printer - a printer which is the instance of the ConsolePrinter class.
35     */
36    void prettyPrint(Printer printer);
37  
38    /**
39     * Get the result to see if a cell was hit.
40     *
41     * @return true if it was hit, false otherwise.
42     */
43    Boolean getIsHit();
44  
45    /**
46     * Updates the cells which belong to the ship using observer pattern.
47     *
48     * @param observable - a ship serves as the observable in observer pattern.
49     */
50    void update(Ship observable);
51  
52  
53  }
